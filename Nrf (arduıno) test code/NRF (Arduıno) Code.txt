#include <SPI.h>
#include <nRF24L01.h>
#include <RF24.h>

RF24 radio(9, 10); // CE, CSN

void setup() {
  Serial.begin(9600);
  Serial.println("NRF24 Test Basliyor...");

  if (!radio.begin()) {
    Serial.println("NRF24 bulunamadi!");
    while (1);
  }

  Serial.println("NRF24 baglandi!");

  radio.setPALevel(RF24_PA_LOW);
  radio.setDataRate(RF24_1MBPS);
  radio.setChannel(108);

  radio.printDetails(); // TÜM bilgileri gösterir
}

void loop() {
  // boş
}
